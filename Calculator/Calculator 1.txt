<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Calculator</title>
</head>
<body>
    <h2>Calculator</h2>
    <label for="num1">Number 1:</label>
    <input type="number" id="num1">
    <br>
    <label for="num2">Number 2:</label>
    <input type="number" id="num2">
    <br>
    <button id="calculateBtn">Calculate</button>
    <br>
    <p id="result"></p>

    <script>
        const num1Input = document.getElementById('num1');
        const num2Input = document.getElementById('num2');
        const calculateBtn = document.getElementById('calculateBtn');
        const resultOutput = document.getElementById('result');

        // Function to calculate multiplication and division
        function calculate() {
            const num1 = parseFloat(num1Input.value);
            const num2 = parseFloat(num2Input.value);

            if (isNaN(num1) || isNaN(num2)) {
                resultOutput.textContent = 'Please enter valid numbers.';
                return;
            }

            const multiplication = num1 * num2;
            const division = num1 / num2;

            resultOutput.textContent = `Multiplication: ${multiplication}, Division: ${division}`;
        }

        // Event listener for button click
        calculateBtn.addEventListener('click', calculate);

        // Event listener for key press (multiplication: *)
        document.addEventListener('keypress', function(event) {
            if (event.key === '*') {
                calculate();
            }
        });

        // Event listener for key press (division: /)
        document.addEventListener('keypress', function(event) {
            if (event.key === '/') {
                calculate();
            }
        });
    </script>
</body>
</html>
